## Replace cell 21 — real Groq-backed interpretive agents

```python
def pathologist_agent(state: GraphState) -> dict:
    vision = state["vision_output"]
    prompt = f"""You are an agricultural plant pathologist assistant. Given this disease
classification JSON from a computer vision model, write a short (3-4 sentence) plain-language
interpretation for a farmer.

Disease classification JSON:
{json.dumps(vision, indent=2)}

Rules:
- Do not invent facts not present in the JSON.
- If is_ambiguous is true, clearly state the model is not confident and a human review is
  needed; do not recommend any treatment action.
- Keep it under 80 words, no jargon.
"""
    resp = groq_client.chat.completions.create(
        model=GROQ_MODEL,
        messages=[{"role": "user", "content": prompt}],
        temperature=0.3,
        max_tokens=200,
    )
    return {"pathologist_interpretation": resp.choices[0].message.content.strip()}


def climate_agent(state: GraphState) -> dict:
    analytical = state["analytical_output"]
    prompt = f"""You are an agronomy climate-risk assistant. Given this 14-day environmental
risk JSON (with SHAP-attributed stressors), write a short (3-4 sentence) plain-language
interpretation for a farmer.

Environmental risk JSON:
{json.dumps(analytical, indent=2)}

Rules:
- Only reference the stressors and risk_band actually present in the JSON.
- Explain which factors are increasing vs decreasing risk, in plain language.
- Keep it under 80 words, no jargon.
"""
    resp = groq_client.chat.completions.create(
        model=GROQ_MODEL,
        messages=[{"role": "user", "content": prompt}],
        temperature=0.3,
        max_tokens=200,
    )
    return {"climate_interpretation": resp.choices[0].message.content.strip()}


def hitl_router(state: GraphState) -> str:
    if state["vision_output"]["is_ambiguous"]:
        return "ambiguous"
    return "proceed"


def ambiguous_review_node(state: GraphState) -> dict:
    return {"status": "AMBIGUOUS_REVIEW_REQUIRED", "route": "ambiguous"}


def proceed_to_translation_node(state: GraphState) -> dict:
    return {"status": "READY_FOR_TRANSLATION_AND_REPORTING", "route": "proceed"}
```

## Update cell 23 — point the graph at the real functions (was `*_stub`)

```python
builder = StateGraph(GraphState)

builder.add_node("vision", vision_node)
builder.add_node("analytical", analytical_node)
builder.add_node("pathologist_agent", pathologist_agent)   # was pathologist_agent_stub
builder.add_node("climate_agent", climate_agent)           # was climate_agent_stub
builder.add_node("ambiguous_review", ambiguous_review_node)
builder.add_node("proceed_to_translation", proceed_to_translation_node)

builder.set_entry_point("vision")
builder.add_edge("vision", "analytical")
builder.add_edge("analytical", "pathologist_agent")
builder.add_edge("pathologist_agent", "climate_agent")

builder.add_conditional_edges(
    "climate_agent",
    hitl_router,
    {"ambiguous": "ambiguous_review", "proceed": "proceed_to_translation"},
)

builder.add_edge("ambiguous_review", END)
builder.add_edge("proceed_to_translation", END)

graph = builder.compile()
print("Graph compiled OK")
```
